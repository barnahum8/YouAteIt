
DROP TABLE IF EXISTS foodTypes;
DROP TABLE IF EXISTS beers;
DROP TABLE IF EXISTS users ;
DROP TABLE IF EXISTS usersFoodTypes;

CREATE TABLE foodTypes (
	id serial PRIMARY KEY,
	name VARCHAR ( 50 ) NOT NULL
);

CREATE TABLE beers (
	id serial PRIMARY KEY,
	name VARCHAR ( 50 ) NOT NULL
);

CREATE TABLE users (
	email VARCHAR ( 50 ) PRIMARY KEY,
	firstname VARCHAR ( 50 ) NOT NULL,
	lastname VARCHAR ( 50 ) NOT NULL,
	birthdate VARCHAR ( 50 ) NOT NULL,
	beer INT FOREIGN KEY REFERENCES beers(id),
	id VARCHAR ( 50 ) NOT NULL,
	phone VARCHAR ( 50 ) NOT NULL
);


CREATE TABLE usersFoodTypes (
	email VARCHAR ( 50 ) FOREIGN KEY REFERENCES users(email),
	id INT NOT NULL,
	FOREIGN KEY (id)
      		REFERENCES foodTypes(id)
	PRIMARY KEY(email,id)
);
